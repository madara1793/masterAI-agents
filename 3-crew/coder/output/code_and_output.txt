# Python program to calculate the first 10,000 terms of the series
# 1 - 1/3 + 1/5 - 1/7 + ... and multiply by 4

def calculate_series(n_terms):
    series_sum = 0
    for i in range(n_terms):
        term = (-1) ** i / (2 * i + 1)  # Generates the terms: 1, -1/3, 1/5, -1/7, ...
        series_sum += term
    return series_sum * 4

# Calculate the sum of the first 10,000 terms
result = calculate_series(10000)
print(result)
